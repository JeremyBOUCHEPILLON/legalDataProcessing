INSERT { graph ?situation { :LED10_explicit_compliant nrv:hasCompliance ?situation .}}
WHERE { 
	{SELECT DISTINCT ?action
		WHERE {
		:LED10_explicit_compliant :isApplicable ?situation .
		?situation :isAbout ?action .
        	?action a :Action .
    		?action :involvesData ?dataset .
    		?dataset :containsData ?data .
		?action :isNecessary "true"^^xsd:boolean .
		{ ?action :isAuthorizedLaw "true"^^xsd:boolean .}
		UNION
		{ ?action :protectsVitalInterests "true"^^xsd:boolean .}
		UNION
		{ FILTER NOT EXISTS {
			?dataset :containsData ?data2 .
			?data2 a :SensitivePersonalData .
			FILTER NOT EXISTS {
				?data2 a :PublicData .
			}
		}}
	}}
	GRAPH ?situation {?action ?p ?v .}
}
