SELECT DISTINCT ?dataset ?data ?datatype ?origin ?destination ?stored
WHERE { 
    ?dataset a lea-proc:Dataset .
    ?dataset lea-proc:containsData ?data .
    ?data a ?datatype .
    OPTIONAL{ ?dataset lea-proc:hasOriginData ?origin .}
    OPTIONAL{ ?dataset lea-proc:hasDestinationData ?destination .}
    OPTIONAL{ ?data lea-proc:isInformationInPossessionOf ?stored .}
}
