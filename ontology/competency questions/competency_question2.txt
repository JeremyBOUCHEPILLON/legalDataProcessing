SELECT DISTINCT ?procedure ?emergency ?authorized ?pending ?protects
WHERE { 
    ?procedure a lea-proc:Procedure .
    ?procedure lea-proc:linkedToAction ?action .
    ?action a lea-proc:Action .
    OPTIONAL{ ?procedure lea-proc:happensInSitationProcedure ?situation .
	      ?situation a ?situationType .}
    OPTIONAL{ ?action lea-proc:isAuthorizedLaw ?authorized .}
    OPTIONAL{ ?action lea-proc:isPendingLegalRemedy ?pending .}
    OPTIONAL{ ?action lea-proc:protectsVitalInterests ?protects .}
}
